            // --- App Store Icon Drawing ---

            function drawAppleIcon() {

                const canvas = document.getElementById('apple-icon');

                if (!canvas) return;

                const ctx = canvas.getContext('2d');

                ctx.scale(1, 1); // Adjusted scale

                ctx.fillStyle = 'white';

                // Apple logo path (simplified)

                ctx.beginPath();

                ctx.arc(16, 10, 3, 0, Math.PI * 2);

                ctx.fill();

                ctx.beginPath();

                ctx.moveTo(10, 30);

                ctx.bezierCurveTo(22, 30, 22, 10, 10, 10);

                ctx.bezierCurveTo(10, 30, 22, 30, 10, 30);

                ctx.fill();

            }


            function drawGooglePlayIcon() {

                const canvas = document.getElementById('google-play-icon');

                if (!canvas) return;

                const ctx = canvas.getContext('2d');

                ctx.scale(1, 1); // Adjusted scale

                // Triangle path

                ctx.beginPath();

                ctx.moveTo(8, 5);

                ctx.lineTo(24, 16);

                ctx.lineTo(8, 27);

                ctx.closePath();

                ctx.fillStyle = 'white';

                ctx.fill();

            }
